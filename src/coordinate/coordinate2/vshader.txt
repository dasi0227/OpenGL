#version 330 core

layout (location = 0) in vec3 VertexCoord;
layout (location = 1) in vec2 VertexTexture;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec2 FragTexture;

void main() {
  gl_Position = projection * view * model * vec4(VertexCoord, 1.0f);
  FragTexture = VertexTexture;
}
